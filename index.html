SessionList — Moniteurs Java

Fidèle au squelette. Indices 0..n/2-1 → profs, n/2..n-1 → étudiants. Utilise synchronized + wait() / notifyAll().
public class SessionList<Session> {
    private ArrayList<Session> buf;
    SessionList(int nbSlots) {
        int i;
        this.buf = new ArrayList<SessionList>(nbSlots);
        for (i=0; i<nbSlots; i++) { buf.add(i,null); }
    }
    public synchronized Session getProfSession() throws InterruptedException {
        int slot=-1;
        while(slot==-1){
            for(int i=0;i<buf.size()/2;i++)
                if(buf.get(i)!=null){slot=i;break;}
            if(slot==-1)wait();
        }
        Session s=buf.get(slot); buf.set(slot,null); notifyAll(); return s;
    }
    public synchronized Session getStudentSession() throws InterruptedException {
        int slot=-1;
        while(slot==-1){
            for(int i=buf.size()/2;i<buf.size();i++)
                if(buf.get(i)!=null){slot=i;break;}
            if(slot==-1)wait();
        }
        Session s=buf.get(slot); buf.set(slot,null); notifyAll(); return s;
    }
    public synchronized void addSession(Session s) throws InterruptedException {
        int debut=s.type.equals("prof")?0:buf.size()/2;
        int fin=s.type.equals("prof")?buf.size()/2:buf.size();
        int slot=-1;
        while(slot==-1){
            for(int i=debut;i<fin;i++)
                if(buf.get(i)==null){slot=i;break;}
            if(slot==-1)wait();
        }
        buf.set(slot,s); notifyAll();
    }
}

SessionList — Sémaphores Java

Même structure. Semaphore de places libres et occupées par zone. mutex protège l'accès à buf.
import java.util.concurrent.Semaphore;
public class SessionList<Session> {
    private ArrayList<Session> buf;
    private final Semaphore mutex=new Semaphore(1);
    private final Semaphore videProf, pleinProf=new Semaphore(0);
    private final Semaphore videStudent, pleinStudent=new Semaphore(0);
    SessionList(int nbSlots){
        int i; this.buf=new ArrayList<SessionList>(nbSlots);
        for(i=0;i<nbSlots;i++){buf.add(i,null);}
        videProf=new Semaphore(nbSlots/2); videStudent=new Semaphore(nbSlots/2);
    }
    public Session getProfSession() throws InterruptedException {
        pleinProf.acquire(); mutex.acquire();
        Session s=null;
        for(int i=0;i<buf.size()/2;i++)
            if(buf.get(i)!=null){s=buf.get(i);buf.set(i,null);break;}
        mutex.release(); videProf.release(); return s;
    }
    public Session getStudentSession() throws InterruptedException {
        pleinStudent.acquire(); mutex.acquire();
        Session s=null;
        for(int i=buf.size()/2;i<buf.size();i++)
            if(buf.get(i)!=null){s=buf.get(i);buf.set(i,null);break;}
        mutex.release(); videStudent.release(); return s;
    }
    public void addSession(Session s) throws InterruptedException {
        boolean p=s.type.equals("prof");
        Semaphore vide=p?videProf:videStudent, plein=p?pleinProf:pleinStudent;
        int debut=p?0:buf.size()/2, fin=p?buf.size()/2:buf.size();
        vide.acquire(); mutex.acquire();
        for(int i=debut;i<fin;i++)
            if(buf.get(i)==null){buf.set(i,s);break;}
        mutex.release(); plein.release();
    }
}

Sans exclusion mutuelle entre types

Deux mutex séparés (mutexProf et mutexStudent) : prof et étudiant peuvent s'exécuter simultanément.
import java.util.concurrent.Semaphore;
public class SessionList<Session> {
    private ArrayList<Session> buf;
    private final Semaphore mutexProf=new Semaphore(1), mutexStudent=new Semaphore(1);
    private final Semaphore videProf, pleinProf=new Semaphore(0);
    private final Semaphore videStudent, pleinStudent=new Semaphore(0);
    SessionList(int nbSlots){
        int i; this.buf=new ArrayList<SessionList>(nbSlots);
        for(i=0;i<nbSlots;i++){buf.add(i,null);}
        videProf=new Semaphore(nbSlots/2); videStudent=new Semaphore(nbSlots/2);
    }
    public Session getProfSession() throws InterruptedException {
        pleinProf.acquire(); mutexProf.acquire();
        Session s=null;
        for(int i=0;i<buf.size()/2;i++)
            if(buf.get(i)!=null){s=buf.get(i);buf.set(i,null);break;}
        mutexProf.release(); videProf.release(); return s;
    }
    public Session getStudentSession() throws InterruptedException {
        pleinStudent.acquire(); mutexStudent.acquire();
        Session s=null;
        for(int i=buf.size()/2;i<buf.size();i++)
            if(buf.get(i)!=null){s=buf.get(i);buf.set(i,null);break;}
        mutexStudent.release(); videStudent.release(); return s;
    }
    public void addSession(Session s) throws InterruptedException {
        if(s.type.equals("prof")){
            videProf.acquire(); mutexProf.acquire();
            for(int i=0;i<buf.size()/2;i++)
                if(buf.get(i)==null){buf.set(i,s);break;}
            mutexProf.release(); pleinProf.release();
        } else {
            videStudent.acquire(); mutexStudent.acquire();
            for(int i=buf.size()/2;i<buf.size();i++)
                if(buf.get(i)==null){buf.set(i,s);break;}
            mutexStudent.release(); pleinStudent.release();
        }
    }
}

VideoCatalog — Priorité aux lecteurs

Fidèle au squelette. Ajout de nbReaders et writing.
public class VideoCatalog {
    private HashMap<String,VideoMetadata> metadata;
    private int nbReaders=0; private boolean writing=false;
    VideoCatalog(){this.metadata=new HashMap<>();}
    public synchronized VideoMetadata readMetadata(String videoId) throws InterruptedException {
        while(writing)wait();
        nbReaders++;notifyAll();
        VideoMetadata result=metadata.get(videoId);
        nbReaders--; if(nbReaders==0)notifyAll(); return result;
    }
    public synchronized void updateMetadata(String videoId,VideoMetadata newData) throws InterruptedException {
        while(writing||nbReaders>0)wait();
        writing=true; metadata.put(videoId,newData); writing=false; notifyAll();
    }
}

VideoCatalog — Priorité aux rédacteurs

Ajout de nbWritersWaiting.
public class VideoCatalog {
    private HashMap<String,VideoMetadata> metadata;
    private int nbReaders=0; private boolean writing=false;
    private int nbWritersWaiting=0;
    VideoCatalog(){this.metadata=new HashMap<>();}
    public synchronized VideoMetadata readMetadata(String videoId) throws InterruptedException {
        while(writing||nbWritersWaiting>0)wait();
        nbReaders++;notifyAll();
        VideoMetadata result=metadata.get(videoId);
        nbReaders--; if(nbReaders==0)notifyAll(); return result;
    }
    public synchronized void updateMetadata(String videoId,VideoMetadata newData) throws InterruptedException {
        nbWritersWaiting++;
        while(writing||nbReaders>0)wait();
        nbWritersWaiting--;writing=true;
        metadata.put(videoId,newData);
        writing=false;notifyAll();
    }
}
